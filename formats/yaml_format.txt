# YAML format for Presentable Object Model

# Parser configuration
parser:
  case_sensitive: false
  ambiguity: resolve
  debug: false

# Terminal definitions
terminals:
  string_format: '"[^"]*" | \'[^\']*\''
  number_format: /[0-9]+(\.[0-9]+)?/
  boolean_format: '"true" | "false" | "yes" | "no" | "on" | "off"'

# Special tokens
special_tokens:
  '#': HASH
  '*': ASTERISK
  '_': UNDERSCORE
  '-': DASH
  '(': LPAREN
  ')': RPAREN
  '[': LBRACK
  ']': RBRACK
  '{': LBRACE
  '}': RBRACE
  ',': COMMA
  ':': COLON
  '.': DOT

# List format - YAML style with indentation
list_format:
  opener: ''  # No bracket
  closer: ''  # No bracket
  separator: '-'  # Dash for list items
  whitespace: true
  indent: true  # Use indentation for lists

# Field clause format
field_clause_template: '{field_name}: {field_value}'

# YAML-specific templates
class_templates:
  LiterateDataModel:
    presentable_template: |
      type: model
      name: {name}
      subjects:
        {subjects}
  
  Subject:
    presentable_template: |
      type: subject
      name: {name}
      level: {level}
      classes:
        {classes}
      subjects:
        {subjects}
  
  Class:
    presentable_template: |
      type: class
      name: {name}
      {{#if one_liner}}
      one_liner: {one_liner}
      {{/if}}
      {{#if is_value_type}}
      is_value_type: true
      {{/if}}
      attributes:
        {attributes}
  
  Attribute:
    presentable_template: |
      type: attribute
      name: {name}
      {{#if one_liner}}
      one_liner: {one_liner}
      {{/if}}
      data_type_clause:
        {data_type_clause}
  
  DataType:
    is_abstract: true
  
  BaseDataType:
    presentable_template: |
      type: base-data-type
      class_: {class_}
      {{#if is_value}}
      is_value: true
      {{/if}}
  
  ListDataType:
    presentable_template: |
      type: list-data-type
      element_type:
        {element_type}
  
  SetDataType:
    presentable_template: |
      type: set-data-type
      element_type:
        {element_type}
  
  MappingDataType:
    presentable_template: |
      type: mapping-data-type
      domain_type:
        {domain_type}
      range_type:
        {range_type}
  
  DataTypeClause:
    presentable_template: |
      data_type:
        {data_type}
      {{#if is_optional}}
      is_optional: true
      {{/if}}

# Boolean values in YAML style
boolean_values:
  is_value_type:
    presentable_true: 'yes'
    presentable_false: 'no'
    explicit: true
  
  is_optional:
    presentable_true: 'yes'
    presentable_false: 'no'
    explicit: false
